{{ config(materialized='table') }}

with limites as (
    select start_time, end_time from {{ ref('stg_iba_time_window') }}
),

series as (
    select generate_series(
        date_trunc('minute', (select start_time from limites)),
        date_trunc('minute', (select end_time from limites)),
        interval '1 minute'
    ) as minute_dt
),

raw_grouped as (
    select 
        date_trunc('minute', data_hora) as minute_dt,
        max(tag_value) as corrente
    from {{ ref('stg_iba_data_raw') }}
    where tag_name = 'II113RC001\U'
      and data_hora between (select start_time from limites) and (select end_time from limites)
    group by 1
),

ffill as (
    select
        s.minute_dt,
        r.corrente,
        max(r.corrente) filter (where r.corrente is not null)
            over (order by s.minute_dt rows between unbounded preceding and current row) as last_non_null
    from series s
    left join raw_grouped r on s.minute_dt = r.minute_dt
)

select
    minute_dt,
    coalesce(corrente, last_non_null) as corrente
from ffill
