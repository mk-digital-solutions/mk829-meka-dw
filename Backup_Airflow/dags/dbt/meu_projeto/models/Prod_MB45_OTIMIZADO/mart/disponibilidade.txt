WITH membrana_duration AS (
  SELECT variables.acum_membrana_on(to_timestamp($__from / 1000), to_timestamp($__to / 1000)) AS t_operacao
),
last_dt AS (
  SELECT dt
  FROM variables.iba_data_raw
  WHERE $__timeFilter(dt) 
  ORDER BY dt DESC
  LIMIT 1
),
tempo_total AS (
  SELECT
    (EXTRACT(EPOCH FROM last_dt.dt) - EXTRACT(EPOCH FROM to_timestamp(${__from} / 1000))) AS t_total
  FROM last_dt
)

SELECT

(SELECT 
  CASE WHEN t_operacao > t_total THEN 100 ELSE (t_operacao / t_total) * 100 END AS disponibiidade
FROM membrana_duration, tempo_total) AS "DISP",

(SELECT variables.acum_eficiencia_prod(
  to_timestamp($__from / 1000),
  to_timestamp($__to / 1000)
)*100 AS eficiencia) AS "EFIC"
